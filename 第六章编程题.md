  ```c++
  #include <iostream>
  #include <fstream>
  #include<cstdlib> //support for exit()
  const int SIZE=60;
  int main() {
      using namespace std;
      cout<<"Enter the charaters and Enter @ to terminated:"<<endl;
      char ch;
      cin.get(ch);
      while(ch!='@')
      {
           if(isupper(ch))
           {
               ch=tolower(ch);
           }
           else if(islower(ch))
           {
               ch=toupper(ch);
           }
           if(!isdigit(ch))
           {
               cout<<ch;
           }
           cin.get(ch);
      }
      cout<<endl;
      return 0;
  }
  ```

  ```c++
  #include <iostream>
  #include <array>
  #include<cstdlib> //support for exit()
  const int SIZE=10;
  using namespace std;
  int main() {
    array<double,SIZE> donations;
    cout<<"Enter donation(no-number to stop):";
    double donation;
    double sum=0.0;
    int i,n;
    for(i=0;(i<SIZE)&&(cin>>donation);i++)
    {
        donations[i]=donation;
        sum+=donation;
    }
    n=i-1;
    //get average
    double average=sum/n;
    cout.precision(4);
    cout<<"n="<<n<<endl;
    cout<<"sum="<<sum<<endl;
    cout<<"Average ="<<average<<endl;
    cout<<"Number larger than average:"<<endl;
    for(i=0;i<n;i++)
    {
        if(donations[i]>average)cout<<donations[i]<<" ";
    }
    cout<<endl;
    return 0;
  }
  ```

  ```c++
  #include <iostream>
  #include <array>
  #include<cstdlib> //support for exit()
  const int SIZE=20;
  using namespace std;
  struct bop
  {
      char fullname[SIZE];
      char title[SIZE];
      char bopname[SIZE];
      int perference;
  };
  int main() {
      int i;
      struct bop array[2];
      cout<<"Input the message:"<<endl;
      for(i=0;i<2;i++)
      {
          cout<<"name:";
          cin>>array[i].fullname;
          cout<<endl;
          cout<<"titile:";
          cin>>array[i].title;
          cout<<endl;
          cout<<"bopname:";
          cin>>array[i].bopname;
          cout<<endl;
          cout<<"perference:";
          cin>>array[i].perference;
          cout<<"----------"<<endl;
      }
      char choice;
      cout<<"a. display by name   "<<endl<<"b. display by title"<<endl<<"c. display by bopname"<<endl<<"d. display by perference"<<endl<<"q. quit"<<endl;
      cin>>choice;
      while(choice!='q')
      {
          switch (choice)
          {
              case 'a':
                  for(int i=0;i<2;i++)
                  {
                      cout<<array[i].fullname<<endl;
                  }
                  break;
              case 'b':
                  for(int i=0;i<2;i++)
                  {
                      cout<<array[i].title<<endl;
                  }
                  break;
              case 'c':
                  for(int i=0;i<2;i++)
                  {
                      cout<<array[i].bopname<<endl;
                  }
                  break;
              case 'd':
                  for(int i=0;i<2;i++)
                  {
                      if(array[i].perference==1)cout<<array[i].fullname<<endl;
                      else if(array[i].perference==2)cout<<array[i].title<<endl;
                      else if(array[i].perference==3)cout<<array[i].bopname<<endl;
                  }
                  break;
              default:
                  cout<<"please enter the right choice.....";
                  break;
          }
          cin>>choice;
      }
      cout<<"bye";
      return 0;
  }
  ```
