- 第二题

  ```c++
  #include <iostream>
  #include <cstring>
  #include <malloc.h>
  using namespace  std;
  typedef  struct
  {
      char s[30];
      double weight;
      int heat;
  } CandyBar;
  void  function(CandyBar &c, const char *s="Millennium",double b=2.85,int i=350)
  {
      strcpy(c.s,s);
      c.weight=b;
      c.heat=i;
  }
  
  void show(const CandyBar &c)
  {
      cout<<"Brand:"<<c.s<<endl<<"weight:"<<c.weight<<endl<<"heat:"<<c.heat<<endl;
  }
  int main()
  {
      CandyBar index;
      char *s ="candy";
      double weight=3.14;
      int heat=36;
  
      cout<<"------------------before-------------------"<<endl;
      function(index);
      show(index);
      function(index,s,weight,heat);
      cout<<"------------------After--------------------"<<endl;
      show(index);
      return 0;
  }
  //问题：为什么function 第二个参数前面是const，为啥还能传值？
  //答案：在于程序先运行main,当main当中给出参数时，const住的值为main传进去的值，如果没有给出参数，则const住的值为默认值。
  ```

- 第三题

  ```c++
  #include <iostream>
  #include <cctype>
  using namespace  std;
  void  to_upper(string &s)
  {
      for(int i=0;i<s.size();i++)
      {
          if(isalpha(s[i]))
          {
              s[i]=toupper(s[i]);
          }
      }
  }
  int main()
  {
       string s;
       cout<<"Input string:"<<endl;
       cin>>s;
       cout<<endl;
       while(s!="q")
       {
           cout<<"Before:"<<s<<endl;
           to_upper(s);
           cout<<"After:"<<s<<endl;
           cout<<"input:";
           cin>>s;
           cout<<endl;
       }
       cout<<"Bye"<<endl;
       return 0;
  }
  //这里使用了cctype当中的转换成大写的函数touuper,同时能可以在ascii码级别对字符进行操作
  //通过该例子发现访问C++当中的string类型中某个元素的方法。
  ```

- 第四题

  ```c++
  #include <iostream>
  #include <cstring>
  
  using namespace std;
  
  struct stringy{
      char * str;
      int ct;
  };
  
  void set(stringy &, char *);
  void show(const stringy &, int n = 1);
  void show(const char *, int n = 1);
  
  int main(void)
  {
      stringy beauy;
      char testing[] = "Reality isn't what it used to be.";
  
      set(beauy, testing);
      show(beauy);
      show(beauy, 3);
      testing[0] = 'D';
      testing[1] = 'u';
      show(testing);
      show(testing, 2);
      show("Done!");
  
      return 0;
  }
  
  void set(stringy & beauy, char * str)
  {
      int len = strlen(str);
  
      beauy.str = new char [len+1];
      strcpy(beauy.str, str);
  
      beauy.ct = len;
  }
  
  void show(const stringy & beauy, int n)
  {
      while(n-- > 0)
          cout << beauy.str << endl;
  }
  
  void show(const char * str, int n)
  {
      while(n-- > 0)
          cout << str << endl;
  }
  ```

- 第五题

  ```c++
  #include <iostream>
  #include <cctype>
  using namespace  std;
  template <typename T>
   T findMAX(T *c)
  {
      T max= c[0];
      for( int i=0;i<5;i++)
      {
           if(c[i]>max)
           {
               max=c[i];
           }
      }
      return max;
  }
  int main()
  {
      int a[5]={1,2,3,4,5};
      double b[5]={1.1,2.1,3.1,4.1,5.1};
      cout<<"MAX(int):"<<findMAX(a)<<endl;
      cout<<"MAX(double)"<<findMAX(b)<<endl;
      return 0;
  }
  ```
